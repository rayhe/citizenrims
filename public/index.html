Fetch Example

fetch('/feed.json')
  .then(r => r.json())
  .then(({ meta, incidents, cases }) => {
    console.log(`${meta.incident_count} incidents, ${meta.case_count} cases`)
    incidents.forEach(i =>
      console.log(i.callType, i.street, i._agency))
  })
